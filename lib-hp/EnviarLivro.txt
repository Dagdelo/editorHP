%%HP: T(3)A(R)F(.);
\<<
@ Abre o menu com os livros para que se possa enviar um deles para outra pessoa

"Enviar"
{{"\<< Voltar" 0.}} Nomes +

IF 1. CHOOSE THEN
	@ Envia o livro selecionado
	IF DUP 0. \=/ THEN
		DUP Nomes SWAP POS
		'Dados' SWAP GET
		2. \->LIST
		'TEMP' STO
		RCLF -33. SF -35. SF
		IFERR 'TEMP' SEND THEN
			@ Ação cancelada pelo usuário
			DROP
		END
		STOF
		'TEMP' PURGE
	ELSE
		DROP
	END
END

\>>
